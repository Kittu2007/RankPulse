This Vercel Edge Function proxies all AI requests server-side, keeping the API key secret:
// api/ai.js
export const config = { runtime: &apos;edge&apos; };
export default async function handler(req) {
  if (req.method !== &apos;POST&apos;) return new Response(&apos;Method not allowed&apos;, { status: 405 });
  try {
    const body = await req.json();
    const res = await fetch(&apos;https://integrate.api.nvidia.com/v1/chat/completions&apos;, {
      method: &apos;POST&apos;,
      headers: {
        &apos;Authorization&apos;: `Bearer ${process.env.NVIDIA_API_KEY}`,
        &apos;Content-Type&apos;: &apos;application/json&apos;,
        &apos;Accept&apos;: body.stream ? &apos;text/event-stream&apos; : &apos;application/json&apos;,
      },
      body: JSON.stringify(body),
    });
    return new Response(res.body, {
      status: res.status,
      headers: { &apos;Content-Type&apos;: res.headers.get(&apos;Content-Type&apos;) || &apos;application/json&apos;, &apos;Access-Control-Allow-Origin&apos;: &apos;*&apos; }
    });
  } catch (e) {
    return new Response(JSON.stringify({ error: e.message }), { status: 500, headers: { &apos;Content-Type&apos;: &apos;application/json&apos; } });
  }
}

11.3 Key Design Rules
2px borders everywhere ΟΔβ²§βÄù no 1px, no rounded corners. RankPulse is brutalist.
Shadow pattern: 4px 4px 0 var(--black) ΟΔβ²§βÄù flat offset shadow, never blur/spread.
Hover state: translate(-1px, -1px) + stronger shadow ΟΔβ²§βÄù the button &apos;lifts&apos;.
All headings: Bebas Neue via var(--font-d). Body: Inter via var(--app-font-sans).
Color discipline: red (#E63946) for ONE primary action per view. Not for decoration.
Score numbers: always Bebas Neue, always large (48px+), either red or white on dark bg.
Section labels: always .label-sm (10px, bold, uppercase, red or #888).
Loading states: button shows &apos;Loading...&apos; text AND disabled attribute.

function cosineSim(a: number[], b: number[]): number {
  const dot = a.reduce((s, v, i) =&gt; s + v * b[i], 0);
  const magA = Math.sqrt(a.reduce((s, v) =&gt; s + v * v, 0));
  const magB = Math.sqrt(b.reduce((s, v) =&gt; s + v * v, 0));
  return dot / (magA * magB);
}

const findSimilar = async (term: string) =&gt; {
  if (!hasAiKey()) return;
  setEmbedLoading(term);
  try {
    const [targetVec, ...otherVecs] = await Promise.all([
      aiEmbed(term),
      ...keywords.filter(k =&gt; k.term !== term).map(k =&gt; aiEmbed(k.term)),

    setSimilarMap(prev =&gt; ({ ...prev, [term]: scored }));
  } finally { setEmbedLoading(null); }
};
